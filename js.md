      const START_WRAPPER_CLASS = "modernized-wrapper-start";
      const END_WRAPPER_CLASS = "modernized-wrapper-end";

      function processModernizedWrappers(root) {
        const starts = Array.from(root.getElementsByClassName(START_WRAPPER_CLASS));
        if (!starts.length) {
          return;
        }

        for (let i = 0; i < starts.length; i++) {
          const startWrapper = starts[i];
          const attrName = startWrapper.dataset.ts15Attribute;
          if (!attrName) {
            continue;
          }

          let current = startWrapper.nextElementSibling;
          let depth = 0;
          while (current) {
            const next = current.nextElementSibling;
            const className = current.className;
            if (className === START_WRAPPER_CLASS) {
              // If another start wrapper => increase depth
              depth++;
            } else if (className === END_WRAPPER_CLASS) {
              if (depth === 0) {
                // This end belongs to the current start
                current.remove();
                break;
              } else {
                depth--;
              }
            } else if (current.tagName !== "SCRIPT") {
              // Apply attribute only to real content nodes
              current.setAttribute(attrName, "");
            }

            current = next;
          }

          // Remove the start wrapper itself
          startWrapper.remove();
        }
      }

      document.addEventListener("DOMContentLoaded", function () {
        // Run once for initial page
        processModernizedWrappers(document);

        // Watch for AJAX DOM changes
        const observer = new MutationObserver(mutations => {
          for (const mutation of mutations) {
            for (const node of mutation.addedNodes) {
              if (node.nodeType !== 1) {
                continue;
              }

              if (node.className === START_WRAPPER_CLASS) {
                processModernizedWrappers(document);
                return;
              }
            }
          }
        });

        observer.observe(document.body, {
          childList: true,
          subtree: true
        });
      });
